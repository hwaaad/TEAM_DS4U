[TABLE] STF - 사원 정보

create table stf(
    STF_ID varchar(10) not null,
    STF_PW varchar(20),
    STF_NM varchar(30),
    STF_PH varchar(13),
    STF_EML varchar(30),
    STF_DEP varchar(20),
    STF_PF varchar(50),
    STF_EMLHASH varchar(64),
    STF_EMLCK tinyint(1),
    primary key(STF_ID));

[TABLE] BOARD - 게시판

create table board(
    STF_ID varchar(10),
    BOARD_SQ int,
    BOARD_TYPE varchar(10),
    BOARD_NM varchar(64),
    BOARD_TXT varchar(255),
    BOARD_DT datetime,
    BOARDHIT int,
    BOARD_FILE varchar(100),
    BOARD_RFILE varchar(100),
    BOARD_GROUP int,
    BOARD_SEQUENCE int,
    BOARD_LEVEL int,
    BOARD_AVAILABLE int,
    primary key(BOARD_SQ));

[TABLE] APV - 정보화 사업
CREATE TABLE APV(
   STF_ID VARCHAR(20),
   APV_SQ int,   
   APV_NM VARCHAR(64),
   APV_DATE VARCHAR(255),
   APV_STT_DATE VARCHAR(64),
   APV_FIN_DATE VARCHAR(64),
   APV_BUDGET VARCHAR(64),   
   APV_PHONE VARCHAR(30),
   APV_POLICY_SQ VARCHAR(30),
   APV_FILE VARCHAR(100),
   APV_RFILE VARCHAR(100),
   APV_GROUP INT,
   APV_SEQUENCE INT,
   primary key (APV_SQ)
);

[TABLE] REQ - 보안성 검토 의뢰
create table REQ(
   STF_ID VARCHAR(20),
   REQ_SQ INT,
   APV_SQ INT,
   APV_NM VARCHAR(30),
   APV_OBJ VARCHAR(600),
   APV_CONT VARCHAR(1000),
   APV_DATE VARCHAR(100),
   REQ_DATE VARCHAR(64),
   REQ_REC_DATE VARCHAR(64),
   REQ_SUB_DATE VARCHAR(64),
   REQ_GROUP INT,
   REQ_SEQUENCE INT,
   PRIMARY KEY(REQ_SQ)
);
